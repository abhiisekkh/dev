pipeline{
    agent any
    stages{
        stage('Checkout')
        {
            steps{
                git branch: 'master' , url: 'https://github.com/drishtii04/p8repo.git'
            }
        }
        stage('Build')
        {
            steps{
                sh 'mvn clean package'
            }
        }
        stage('Test')
        {
            steps{
                sh 'mvn test'
            }
        }
        stage('Archive Artifact')
        {
            steps{
                archiveArtifacts artifacts: '**/target/*.jar',
                allowEmptyArchive: true
            }
        }
        stage('Deploy')
        {
            steps{
                sh """
                export ANSIBLE_HOST_KEY_CHECKING=False
                ansible-playbook -i host.ini p8.yml --extra-vars='ansible_become_pass=2004'
                """
            }
        }
    }
}














---
- name: deploy maven artifact
  hosts: localhost
  tasks:
   - name: Ansible
     become: true
     become_user: drishti
     become_method: su
     copy:
      src: "/var/lib/jenkins/workspace/p8pipe/target/prog8-1.0-SNAPSHOT.jar"
      dest: "/home/drishti/059/prog8/p8.jar"
